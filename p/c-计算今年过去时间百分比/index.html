C++ 计算今年过去时间百分比

Oct 15, 2018
3 minute read

昨晚在 App Store 看到了一款应用，这个应用提供了一个 widget 来显示今年已经过去了多少，提醒你珍惜自己的时间，每过去大约 52 分钟，百分比就会增长 0.01%，但是要付费，但是 3 块钱不贵，我就直接买了支持一下作者，看了作者头像是只猫，恩看来很多程序员是喜欢猫的 :)

时间规划局

emmmm，还行还行，主要计算算法如下：

一天 24 个小时，一小时 60 分钟，一分钟 60 秒

所以一天有：24×60×60＝86400 秒

一年总秒数：总天数 * 24 * 60 * 60

所以最后可以通过这个公式算出：

一年过去的秒数 ÷ 一年的总秒数 × 100.0

#include <iostream>
#include <ctime>

int main(int argc, char *argv[])
{
    time_t now = time(0);
    tm *ltm = localtime(&now);
    int year = 1900 + ltm->tm_year;
    int total_days = 0;

    // 公历是 4 的倍数就是闰年，例如 2010 除以 4,不是整数倍,所以 2010 年是平年.
    if (year % 400 == 0 || (year % 4 == 0 && year % 100 != 0)) {
        total_days = 366;
    } else {
        total_days = 365;
    }

    long total_seconds = total_days * 24 * 60 * 60;

    // 年初时间
    struct tm year_time;
    year_time.tm_hour = 00;
    year_time.tm_min = 00;
    year_time.tm_sec = 00;
    year_time.tm_mday = 01;
    year_time.tm_mon = 1-1;
    year_time.tm_year = ltm->tm_year;

    // 算出今年年初到现在的秒数
    double now_seconds = difftime(mktime(ltm), mktime(&year_time));

    double percent = (double)now_seconds / (double)total_seconds * 100.0;

    std::cout << percent << std::endl;

    return 0;
}

#include <QApplication>
#include <QVBoxLayout>
#include <QWidget>
#include <QLabel>
#include <QTimer>

int main(int argc, char *argv[])
{
    QApplication a(argc, argv);

    QWidget *window = new QWidget;
    QLabel *label = new QLabel;
    QVBoxLayout *mainLayout = new QVBoxLayout(window);
    mainLayout->addWidget(label, 0, Qt::AlignCenter);

    QTimer *timer = new QTimer;

    QObject::connect(timer, &QTimer::timeout, [=] {
        time_t now = time(0);
        tm *ltm = localtime(&now);
        int year = 1900 + ltm->tm_year;
        int total_days = 0;

        if (year % 400 == 0 || (year % 4 == 0 && year % 100 != 0)) {
            total_days = 366;
        } else {
            total_days = 365;
        }

        long long total_seconds = total_days * 24 * 60 * 60;

        struct tm year_time;
        year_time.tm_hour = 00;
        year_time.tm_min = 00;
        year_time.tm_sec = 00;
        year_time.tm_mday = 01;
        year_time.tm_mon = 1-1;
        year_time.tm_year = ltm->tm_year;

        long long now_seconds = difftime(mktime(ltm), mktime(&year_time));
        double percent = now_seconds * 1.0 / total_seconds * 100.0;

        label->setText(QString("%1 年已过去 %2").arg(year).arg(QString::number(percent, 'f', 6)));
    });

    timer->setInterval(100);
    timer->start();

    window->resize(300, 300);
    window->show();

    return a.exec();
}

class Application : GLib.Object {
    static int main(string[] args) {
        time_t time_stamp = time_t();
        var t = GLib.Time.local(time_stamp);
        int year = t.year + 1900;
        // Years since 1900（1900 年以来，所以需要加上 1900）
        int total_days = 0;

        if (year % 400 == 0 || (year % 4 == 0 && year % 100 != 0)) {
            total_days = 366;
        } else {
            total_days = 365;
        }

        long total_second = total_days * 24 * 60 * 60;
        long now_second = t.day_of_year * 24 * 60 * 60;
        double percent = now_second * 1.0 / total_second * 100;

        stdout.printf("总秒数：%ld\n", total_second);
        stdout.printf("现秒数：%ld\n", now_second);
        stdout.printf("%d 已过去 %f%\n", year, percent);

        return 0;
    }
}

最后

两个 long 类型的值相除时，结果会自动取整，可以在被除数乘以 1.0 再除以除数，这样得到的结果就不会取整为零了。

当然也可以强转为 double 类型，不过不推荐这样的做法。
